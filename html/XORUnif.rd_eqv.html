Library XORUnif.rd_eqv

From XORUnif Require Export Reduced.

Definition rd_eqv (t1 t2:lTerm):Prop:=
  Reducing_lr_Ord t1 = Reducing_lr_Ord t2.

Definition rd_eqv_bool (t1 t2:lTerm):bool:=
  lTerm_eqv_bool t1 t2.

Lemma rd_eqv_Prop_bool: ∀(t1 t2:lTerm),
  rd_eqv t1 t2 ↔ rd_eqv_bool t1 t2 = true.
Proof.
  split;intros; unfold rd_eqv in *; unfold rd_eqv_bool in *;unfold lTerm_eqv_bool in ×.
  - rewrite H. apply term_beq_syn_list_correct. auto.
  - apply term_beq_syn_list_correct in H. auto.
Qed.

Notation "x =+= y" := (rd_eqv x y) (at level 70).

Lemma rd_eqv_relf:∀ (lt:lTerm),
 lt =+= lt.
Proof.
  intros. unfold rd_eqv. auto.
Qed.

Lemma rd_eqv_sym:∀ (lt1 lt2:lTerm),
 lt1 =+= lt2 → lt2 =+= lt1.
Proof.
  intro;unfold rd_eqv; auto.
Qed.

Lemma rd_eqv_trans:∀ (lt1 lt2 lt3:lTerm),
 lt1 =+= lt2 → lt2 =+= lt3 → lt1 =+= lt3.
Proof.
  intro;unfold rd_eqv; intros. congruence.
Qed.

Add Parametric Relation : lTerm rd_eqv
    reflexivity proved by @rd_eqv_relf
    symmetry proved by @rd_eqv_sym
    transitivity proved by @rd_eqv_trans
      as rd_eq_set_rel.

Definition oplus_lTerm (lt1 lt2:lTerm): lTerm:=
  lt1 ++ lt2.

Lemma rd_eqv_compat:∀ (lt1 lt2 lt3 lt4:lTerm),
  lt1 =+= lt2 → lt3 =+= lt4 →
  oplus_lTerm lt1 lt3 =+= oplus_lTerm lt2 lt4.
Proof.
  intros. unfold rd_eqv in ×.
  apply Reducing_lr_Ord_compat;auto.
Qed.

Add Parametric Morphism : oplus_lTerm with
      signature rd_eqv ==> rd_eqv ==> rd_eqv as rd_eqv_mor.
Proof.
  intros. apply rd_eqv_compat;auto.
Qed.

Lemma rd_eqv_compat':∀ (lt1 lt2 lt3 lt4:lTerm),
  lt1 =+= lt2 → lt3 =+= lt4 →
  (lt1 ++ lt3) =+= lt2 ++ lt4.
Proof.
  intros. rewrite H. rewrite H0. apply rd_eqv_relf.
Qed.

Lemma rd_eqv_f_equal: ∀(lt1 lt2:lTerm)(t:term),
  lt1 =+= lt2 → t::lt1 =+= t::lt2.
Proof.
  intros. unfold rd_eqv in ×. assert(L:=Reducing_lr_Ord_compat [t] [t] lt1 lt2).
  simpl in L. apply L;auto.
Qed.

Lemma Permutation_rd_eqv:∀(lt1 lt2:lTerm),
  Permutation lt1 lt2 → lt1 =+= lt2.
Proof.
  intros. induction H.
  - reflexivity.
  - apply rd_eqv_f_equal. auto.
  - assert(L:=Reducing_lr_Ord_compat [y;x] [x;y] l l). simpl in L.
    apply L;auto. rewrite <- Reducing_lr_Ord_oplus. rewrite <- Reducing_lr_Ord_oplus.
    apply Reducing_lr_Ord_term_swap.
  - apply rd_eqv_trans with l';auto.
Qed.

Lemma Reducing_lr_Ord_nil:
  Reducing_lr_Ord [] = [].
Proof.
  unfold Reducing_lr_Ord. rewrite Reducing_lr_nil. auto.
Qed.

Lemma rd_eqv_N:∀(lt:lTerm),
   lt ++ lt =+= [].
Proof.
  intros. unfold rd_eqv. rewrite Reducing_lr_Ord_nil.
  unfold Reducing_lr_Ord. unfold Reducing_lr. rewrite <- AReducing_lr_app.
  unfold NReducing. rewrite NReducing'_listsame. auto.
Qed.
